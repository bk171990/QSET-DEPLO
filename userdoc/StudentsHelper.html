def admission1_error!
  return if @student.errors.empty?
  messages = @student.errors.full_messages.map { |m| content_tag(:li, m) }.join
  html = <<-HTML
   <div class="alert alert-danger alert-dismissable">
    <ul>#{messages}</ul>
  </div>
  HTML
  html.html_safe
end
