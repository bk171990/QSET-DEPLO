def timetable_error!
  return if  @time_table.errors.empty?
  messages = @time_table.errors.full_messages.map      { |m| content_tag(:li, m) }.join
  html = <<-HTML
   <div class="alert alert-danger alert-dismissable">
    <ul>#{messages}</ul>
  </div>
  HTML
  html.html_safe
end
