def self.search_user(search)
  return if search.empty?
  where("concat_ws(' ',first_name,last_name)like ? \
    OR concat_ws(' ',last_name,first_name)like ? \
    OR username like ?", "#{search}%", "#{search}%", "#{search}%")
end
