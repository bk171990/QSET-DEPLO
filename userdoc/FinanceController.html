def expense_list
  @start_date = params[:expense][:start_date].to_date
  @end_date = params[:expense][:end_date].to_date
  if @end_date.nil? || @start_date.nil?
    flash[:alert] = t('expense_error')

  authorize! :read, @expenses.first unless @expenses.nil?
end
