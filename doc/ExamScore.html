def marks_cant_be_greater_than_maximum_marks
  unless exam.exam_group.exam_type == 'Grades'
    if  marks.present? && marks > exam.maximum_marks
      errors.add(:marks, 'cannot be greater than maximum marks')
    end
  end
end
