def batch_error!
  return if  @batch.errors.empty?
  messages = @batch.errors.full_messages.map { |m| content_tag(:li, m) }.join
  html = <<-HTML
   <div class="alert alert-danger alert-dismissable">
    <ul>#{messages}</ul>
  </div>
  HTML
  html.html_safe
end
