def self.select_employee(e)
  weekdays, class_timings, employees = [], [], []
  unless e.nil?
    e.each do |t|
      weekdays << t.weekday
      class_timings << t.class_timing
      employees << t.employee
    end
  end
  [weekdays, class_timings, employees]
end

def self.select_time_table(time_table)

      batches.push t.batch
    end
    batches
  end
end
