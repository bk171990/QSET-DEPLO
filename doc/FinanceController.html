def view_payslip
  @payslips = []
  @department = EmployeeDepartment.shod(params[:payslip][:department])
  @date = params[:payslip][:date]
  @employees ||= @department.employees
  return if @employees.nil?
  @employees.each do |e|
    salary = e.salary(@date)
    @payslips << salary unless salary.nil?
  end
