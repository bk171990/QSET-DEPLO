def general_setting_error!
  return if  @general_setting.errors.empty?
  msg = @general_setting.errors.full_messages.map \
        { |m| content_tag(:li, m) }.join
  html = <<-HTML
   <div class="alert alert-danger alert-dismissable">
    <ul>#{msg}</ul>
  </div>
  HTML
  html.html_safe
end
