def common_to_all
  Batch.all.each do |batch|
    BatchEvent.create(batch_id: batch.id, event_id: id)
  end
  EmployeeDepartment.all.each do |dept|
    EmployeeDepartmentEvent.create(employee_department_id: \
      dept.id, event_id: id)
  end
end
