def self.set_day(batch, weekdays)
  present_weekdays = Weekday.where(batch_id: batch)
  present_weekdays.destroy_all unless present_weekdays.nil?
  weekdays.each  do |w|
    Weekday.create(batch_id: batch, weekday: day[w.to_i])
  end
  Batch.shod(batch)
end
